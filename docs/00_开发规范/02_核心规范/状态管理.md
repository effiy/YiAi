# 状态管理规范

本文档定义 YiAi 项目的状态管理规范。

---

## 状态管理原则

### 无状态服务

服务应该尽量保持无状态：

```python
# 正确示例 - 无状态
class UserService:
    async def get_user(self, user_id: str):
        return await db.find_one("users", {"id": user_id})

# 避免示例 - 有状态
class UserService:
    def __init__(self):
        self.cache = {}  # 实例状态

    async def get_user(self, user_id: str):
        if user_id in self.cache:
            return self.cache[user_id]
        user = await db.find_one("users", {"id": user_id})
        self.cache[user_id] = user
        return user
```

### 使用外部存储

需要状态时使用外部存储（Redis、数据库等）：

```python
from core.redis import get_redis

class UserService:
    async def get_user(self, user_id: str):
        redis = await get_redis()
        cached = await redis.get(f"user:{user_id}")
        if cached:
            return json.loads(cached)

        user = await db.find_one("users", {"id": user_id})
        if user:
            await redis.setex(f"user:{user_id}", 3600, json.dumps(user))
        return user
```

## 请求状态

### 请求上下文

使用 FastAPI 的请求上下文传递状态：

```python
from fastapi import Request, Depends

async def get_current_user(request: Request) -> Optional[User]:
    """从请求中获取当前用户"""
    return getattr(request.state, "user", None)

@router.get("/profile")
async def get_profile(current_user: User = Depends(get_current_user)):
    return current_user
```

### 中间件设置状态

在中间件中设置请求状态：

```python
@app.middleware("http")
async def auth_middleware(request: Request, call_next):
    token = request.headers.get("X-Token")
    if token:
        user = await verify_token(token)
        request.state.user = user
    response = await call_next(request)
    return response
```

## 全局状态

### 单例模式

使用单例模式管理全局状态（如数据库连接）：

```python
# src/core/database.py
from motor.motor_asyncio import AsyncIOMotorClient

_db_instance = None

class Database:
    client: AsyncIOMotorClient = None
    db = None

db = Database()

async def initialize():
    global _db_instance
    if db.client is None:
        db.client = AsyncIOMotorClient(settings.mongodb_url)
        db.db = db.client[settings.mongodb_db_name]
```

## 会话状态

### 用户会话

使用数据库存储用户会话：

```python
class SessionService:
    async def create_session(self, user_id: str) -> str:
        """创建会话"""
        session_id = str(uuid.uuid4())
        session_data = {
            "id": session_id,
            "user_id": user_id,
            "created_at": datetime.utcnow(),
            "expires_at": datetime.utcnow() + timedelta(days=7)
        }
        await db.insert_one("sessions", session_data)
        return session_id

    async def get_session(self, session_id: str) -> Optional[dict]:
        """获取会话"""
        session = await db.find_one("sessions", {"id": session_id})
        if session and session["expires_at"] > datetime.utcnow():
            return session
        return None
```

## 缓存策略

### 缓存层级

使用多级缓存提升性能：

```python
class CacheService:
    async def get(self, key: str):
        """获取缓存"""
        # 1. 先查本地缓存
        if key in _local_cache:
            return _local_cache[key]

        # 2. 再查 Redis
        redis = await get_redis()
        data = await redis.get(key)
        if data:
            _local_cache[key] = json.loads(data)
            return _local_cache[key]

        return None

    async def set(self, key: str, value: Any, ttl: int = 3600):
        """设置缓存"""
        _local_cache[key] = value
        redis = await get_redis()
        await redis.setex(key, ttl, json.dumps(value))
```

### 缓存失效

合理设置缓存失效策略：

```python
class UserService:
    async def update_user(self, user_id: str, data: dict):
        """更新用户"""
        # 更新数据库
        await db.update_one("users", {"id": user_id}, {"$set": data})

        # 清除相关缓存
        cache = CacheService()
        await cache.delete(f"user:{user_id}")
        await cache.delete_pattern("users:*")
```

## 状态一致性

### 乐观锁

使用乐观锁处理并发更新：

```python
async def update_with_version(user_id: str, data: dict, version: int):
    """使用版本号进行乐观锁更新"""
    result = await db.update_one(
        "users",
        {"id": user_id, "version": version},
        {"$set": data, "$inc": {"version": 1}}
    )
    if result.modified_count == 0:
        raise BusinessException(ErrorCode.CONFLICT, "数据已被修改")
```

### 事务

使用数据库事务保证一致性：

```python
async def transfer_funds(from_id: str, to_id: str, amount: float):
    """转账操作 - 使用事务"""
    session = await db.start_session()
    try:
        async with session.start_transaction():
            # 扣款
            await db.update_one(
                "accounts",
                {"id": from_id, "balance": {"$gte": amount}},
                {"$inc": {"balance": -amount}},
                session=session
            )
            # 加款
            await db.update_one(
                "accounts",
                {"id": to_id},
                {"$inc": {"balance": amount}},
                session=session
            )
    except Exception as e:
        await session.abort_transaction()
        raise
```
