# 错误处理规范

本文档定义 YiAi 项目的错误处理规范。

---

## 异常体系

### 自定义异常

项目使用自定义业务异常：

```python
# src/core/exceptions.py
class BusinessException(Exception):
    """业务异常"""

    def __init__(self, code: int, message: str = None):
        self.code = code
        self.message = message or str(code)
        super().__init__(self.message)
```

### 错误码定义

使用统一的错误码：

```python
# src/core/error_codes.py
from enum import IntEnum

class ErrorCode(IntEnum):
    """错误码枚举"""
    SUCCESS = 0
    INTERNAL_ERROR = 10001
    INVALID_PARAMS = 10002
    NOT_FOUND = 10003
    DATA_ALREADY_EXISTS = 10004
    DATA_DESTROY_FAIL = 10005
```

## 异常抛出规范

### 业务异常

使用 BusinessException 抛出业务错误：

```python
# 正确示例
from core.exceptions import BusinessException
from core.error_codes import ErrorCode

async def get_user(user_id: str):
    user = await db.find_one(settings.collection_sessions, {"id": user_id})
    if not user:
        raise BusinessException(ErrorCode.NOT_FOUND, message="用户不存在")
    return user

async def create_user(user_data: dict):
    existing = await db.find_one(settings.collection_sessions, {"email": user_data["email"]})
    if existing:
        raise BusinessException(ErrorCode.DATA_ALREADY_EXISTS, message="邮箱已被注册")
    # 创建用户...
```

### 参数验证异常

使用 Pydantic 验证或手动验证：

```python
# Pydantic 验证
class UserCreateRequest(BaseModel):
    name: str = Field(..., min_length=1, max_length=100)
    email: EmailStr = Field(...)

# 或手动验证
if not user_id or len(user_id) > 100:
    raise BusinessException(ErrorCode.INVALID_PARAMS, message="无效的用户 ID")
```

### 系统异常

系统异常可以直接抛出，或包装为业务异常：

```python
# 直接抛出 - 让全局异常处理器处理
try:
    await db.insert_one(...)
except Exception as e:
    logger.error(f"数据库操作失败: {e}", exc_info=True)
    raise

# 或包装为业务异常
try:
    await db.insert_one(...)
except Exception as e:
    logger.error(f"数据库操作失败: {e}", exc_info=True)
    raise BusinessException(ErrorCode.INTERNAL_ERROR, message="保存失败") from e
```

## 异常捕获规范

### 捕获特定异常

只捕获预期的异常：

```python
# 正确示例
try:
    result = await some_operation()
except BusinessException:
    # 业务异常，直接抛出
    raise
except ValueError as e:
    # 值错误
    logger.warning(f"参数错误: {e}")
    raise BusinessException(ErrorCode.INVALID_PARAMS) from e
except aiohttp.ClientError as e:
    # HTTP 客户端错误
    logger.error(f"HTTP 请求失败: {e}", exc_info=True)
    raise BusinessException(ErrorCode.INTERNAL_ERROR, message="外部服务调用失败") from e
except Exception as e:
    # 其他异常
    logger.error(f"未知错误: {e}", exc_info=True)
    raise BusinessException(ErrorCode.INTERNAL_ERROR) from e

# 避免示例 - 捕获所有异常并忽略
try:
    result = await some_operation()
except:
    pass  # 不要这样做！
```

### 异常链

保留异常链：

```python
# 正确示例
try:
    await db_operation()
except Exception as e:
    logger.error(f"操作失败: {e}", exc_info=True)
    raise BusinessException(ErrorCode.INTERNAL_ERROR) from e

# 避免示例 - 丢失原始异常
try:
    await db_operation()
except Exception as e:
    logger.error(f"操作失败: {e}", exc_info=True)
    raise BusinessException(ErrorCode.INTERNAL_ERROR)
```

## 全局异常处理

### 异常处理器

项目配置了全局异常处理器：

```python
# src/core/exception_handler.py
def register_exception_handlers(app: FastAPI):
    """注册全局异常处理器"""

    @app.exception_handler(BusinessException)
    async def business_exception_handler(request: Request, exc: BusinessException):
        """处理业务异常"""
        logger.warning(f"业务异常: code={exc.code}, message={exc.message}")
        return JSONResponse(
            status_code=status.HTTP_200_OK,
            content={"code": exc.code, "message": exc.message, "data": None}
        )

    @app.exception_handler(RequestValidationError)
    async def validation_exception_handler(request: Request, exc: RequestValidationError):
        """处理请求验证异常"""
        logger.warning(f"参数验证失败: {exc.errors()}")
        return JSONResponse(
            status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
            content={
                "code": ErrorCode.INVALID_PARAMS,
                "message": "参数验证失败",
                "data": exc.errors()
            }
        )

    @app.exception_handler(Exception)
    async def general_exception_handler(request: Request, exc: Exception):
        """处理未捕获的异常"""
        logger.error(f"未处理的异常: {exc}", exc_info=True)
        return JSONResponse(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            content={
                "code": ErrorCode.INTERNAL_ERROR,
                "message": "服务器内部错误",
                "data": None
            }
        )
```

## 日志记录规范

### 异常日志

异常时记录完整堆栈：

```python
# 正确示例
try:
    await risky_operation()
except BusinessException as e:
    # 业务异常，记录警告
    logger.warning(f"业务异常: code={e.code}, message={e.message}")
    raise
except Exception as e:
    # 系统异常，记录错误和堆栈
    logger.error(f"系统异常: {e}", exc_info=True)
    raise

# 避免示例 - 不记录堆栈
try:
    await risky_operation()
except Exception as e:
    logger.error(f"系统异常: {e}")  # 缺少 exc_info=True
    raise
```

### 日志级别

根据异常类型使用适当的日志级别：

```python
# 业务异常 - WARNING
except BusinessException as e:
    logger.warning(f"业务异常: {e}")
    raise

# 参数验证失败 - WARNING
except ValueError as e:
    logger.warning(f"参数错误: {e}")
    raise

# 系统错误 - ERROR
except Exception as e:
    logger.error(f"系统错误: {e}", exc_info=True)
    raise
```

## 统一响应格式

### 成功响应

使用 `success()` 函数：

```python
from core.response import success

# 基本成功响应
return success(data=user_data)

# 带消息的成功响应
return success(data=user_data, message="创建成功")

# 响应格式
{
  "code": 0,
  "message": "success",
  "data": { ... }
}
```

### 错误响应

使用 `error()` 函数或让异常处理器处理：

```python
from core.response import error
from core.error_codes import ErrorCode

# 返回错误响应
return error(code=ErrorCode.NOT_FOUND, message="用户不存在")

# 或抛出异常（推荐）
raise BusinessException(ErrorCode.NOT_FOUND, message="用户不存在")

# 响应格式
{
  "code": 10003,
  "message": "用户不存在",
  "data": null
}
```

## 错误处理最佳实践

### 1. 早返回，早失败

```python
# 正确示例
async def process_user(user_id: str):
    user = await get_user(user_id)
    if not user:
        raise BusinessException(ErrorCode.NOT_FOUND)

    if not user.get("active"):
        raise BusinessException(ErrorCode.INVALID_PARAMS, message="用户已禁用")

    # 继续处理...
    return result

# 避免示例
async def process_user(user_id: str):
    user = await get_user(user_id)
    if user:
        if user.get("active"):
            # 处理...
            return result
        else:
            raise BusinessException(...)
    else:
        raise BusinessException(...)
```

### 2. 提供有意义的错误信息

```python
# 正确示例
if not file_id:
    raise BusinessException(ErrorCode.INVALID_PARAMS, message="file_id 不能为空")
if len(file_id) > 100:
    raise BusinessException(ErrorCode.INVALID_PARAMS, message="file_id 长度不能超过 100")

# 避免示例
if not file_id or len(file_id) > 100:
    raise BusinessException(ErrorCode.INVALID_PARAMS, message="参数错误")
```

### 3. 使用异常而不是返回错误码

```python
# 正确示例
async def get_user(user_id: str) -> User:
    user = await db.find_one(...)
    if not user:
        raise BusinessException(ErrorCode.NOT_FOUND)
    return user

# 避免示例
async def get_user(user_id: str) -> tuple[Optional[User], Optional[str]]:
    user = await db.find_one(...)
    if not user:
        return None, "用户不存在"
    return user, None
```
